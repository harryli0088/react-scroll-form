## Install

```bash
npm install --save react-scroll-form
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-scroll-form'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
